Setup Git
$ git config --global user.name "<name>"
$ git config --global user.email <email>

Initialize git repo
$ git init

Project Status
$ git status

Add file to stage
$ git add <filename>

Commit file
$ git commit

	options 
	-m "message" //add message to commit
	-a 	     //add files to stage

Check repo history
$ git log
	options
	--graph		//branch graph
	--oneline	
	-p		//changes
	-1 		//number of commits
	--decorate

Create alias
$ git config --global alias.lg 'log --graph -- oneline --decorate -- all'	 
Differences 
$ git diff
	options
	--staged		// changes between stage and last commit

Reset local changes
$ git checkout -- <filename> //reset local changes on file
	                
	          	              
Reset the stage
$git reset HEAD
	options
	<filename>      //unstage file
Remove commit
$ git reset <IDcommit>
	options
	--hard 		//remove changes

Amend last commit
$ git commit --amend

Create Branch
$ git branch <branchname>

Change the branch
$ git checkout <branchname>

Merge branch
$ git merge <branchname>
